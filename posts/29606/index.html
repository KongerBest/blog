制作html烟花特效

2020-01-13 教程 html 0 评论

教你制作炫酷的html烟花特效

​ 想在html中做出烟花特效吗？

​ 想让自己的网页更加出色吗？

​ 你需要”他“！

让我们开始吧！

  • 首先，在桌面上新建一个文本文档，文档名随意。

  • 然后，进入文档编辑，把下列代码复制进去。

  • <!doctype html>
<html lang="en">
 <head>
  <title>SUPER FIREWORKS!</title>
 </head>
 <body>
  <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Fireworks</title>
    <style type="text/css">
    body{
      background:#000;
      margin:0;
      width:100%;
      overflow:hidden;
    }
    </style>
</head>
<body>

<canvas>#FFFFFF</canvas>

    <script>
        alert("BE READY TO BE SUPRISED~");
        var t = 0;
        var c = document.querySelector("canvas");     
        var $ = c.getContext('2d');
        c.width = window.innerWidth;
        c.height = window.innerHeight;
        $.fillStyle = 'hsla(0,0%,0%,1)';
        window.addEventListener('resize', function() {
          c.width = window.innerWidth;
          c.height = window.innerHeight;
        }, false);
        function draw() {
          $.globalCompositeOperation = 'source-over';
          $.fillStyle = 'hsla(0,0%,0%,.1)';
          $.fillRect(0, 0, c.width, c.height);
          var foo, i, j, r;
          foo = Math.sin(t) * 2 * Math.PI;
          for (i=0; i<400; i++) {
            r = 400 * Math.sin(i * foo);
            $.globalCompositeOperation = '';
            $.fillStyle = 'hsla(' + i + 12 + ',100%, 60%,1)';
            $.beginPath();
            $.arc(Math.sin(i) * r + (c.width / 2), 
                  Math.cos(i) * r + (c.height / 2), 
                  1.5, 0, Math.PI * 2);
            $.fill();

          }
          //this controls the effect
          //      ↓↓↓
          t += 0.000005;
          return t %= 2 * Math.PI;//MATH.PI：π
        };

        function run() {
          window.requestAnimationFrame(run);
          draw();
        }
        run();
    </script>
</body>
</html>
 </body>
</html>

  • 退出保存，文件后缀改为 .html

  • 打开欣赏美妙的烟花吧
